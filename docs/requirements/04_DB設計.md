# DB 設計書

**バージョン**: 1.0  
**作成日**: 2026年4月30日  
**DB**: PostgreSQL / Neon（既存アカウントに `travelapp` DB を追加）

---

## 1. テーブル一覧

| テーブル名 | 概要 | 備考 |
|-----------|------|------|
| `airports` | 空港マスター | 起動時に IATA データをシードで投入 |
| `flight_search_cache` | フライト検索結果キャッシュ | 30分で期限切れ |
| `price_calendar_cache` | 価格カレンダーキャッシュ | 24時間で期限切れ |
| `inspire_cache` | どこでも検索結果キャッシュ | 6時間で期限切れ |
| `sale_events` | セールイベント | RSS・Gemini・Affiliate で収集 |
| `favorites` | お気に入り路線 | 個人利用のため user_id 不要 |
| `price_alerts` | 価格アラート設定 | favorites に紐づく |
| `push_subscriptions` | Web Push 購読情報 | デバイスの endpoint を保存 |

## 3. テーブル定義（DDL）

### 3.1 airports

```sql
CREATE TABLE airports (
    iata_code     CHAR(3)        PRIMARY KEY,
    icao_code     CHAR(4),
    name_ja       VARCHAR(100)   NOT NULL,
    name_en       VARCHAR(100)   NOT NULL,
    city_ja       VARCHAR(100),
    city_en       VARCHAR(100),
    country_code  CHAR(2)        NOT NULL,
    country_ja    VARCHAR(100),
    latitude      DECIMAL(9, 6),
    longitude     DECIMAL(9, 6),
    timezone      VARCHAR(50),
    is_active     BOOLEAN        NOT NULL DEFAULT TRUE,
    updated_at    TIMESTAMPTZ    NOT NULL DEFAULT NOW()
);

-- 主要空港の初期データはシードファイルで投入 (scripts/seed_airports.py)
```

### 3.2 flight_search_cache

フライト検索 API のレスポンスを丸ごと JSONB で保存する。  
同じ検索条件（ハッシュが一致）なら API を呼び出さずキャッシュを返す。

```sql
CREATE TABLE flight_search_cache (
    id                  UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    search_query_hash   VARCHAR(64)    NOT NULL UNIQUE,  -- SHA-256(正規化パラメータ)
    origin_iata         CHAR(3)        NOT NULL REFERENCES airports(iata_code),
    destination_iata    CHAR(3)        NOT NULL REFERENCES airports(iata_code),
    departure_date      DATE           NOT NULL,
    return_date         DATE,                            -- NULL = 片道
    cabin_class         VARCHAR(20)    NOT NULL DEFAULT 'ECONOMY',
    adults              SMALLINT       NOT NULL DEFAULT 1,
    nonstop_only        BOOLEAN        NOT NULL DEFAULT FALSE,
    result_json         JSONB          NOT NULL,         -- Amadeus / Kiwi レスポンス
    source_api          VARCHAR(20)    NOT NULL,         -- 'amadeus' | 'kiwi'
    result_count        SMALLINT,
    cached_at           TIMESTAMPTZ    NOT NULL DEFAULT NOW(),
    expires_at          TIMESTAMPTZ    NOT NULL           -- cached_at + interval '30 minutes'
);

CREATE INDEX idx_fsc_hash      ON flight_search_cache(search_query_hash);
CREATE INDEX idx_fsc_expires   ON flight_search_cache(expires_at);
CREATE INDEX idx_fsc_route     ON flight_search_cache(origin_iata, destination_iata, departure_date);
```

### 3.3 price_calendar_cache

指定路線の日別最安値を格納。ヒートマップ表示に使用。

```sql
CREATE TABLE price_calendar_cache (
    id                  UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    origin_iata         CHAR(3)        NOT NULL REFERENCES airports(iata_code),
    destination_iata    CHAR(3)        NOT NULL REFERENCES airports(iata_code),
    travel_date         DATE           NOT NULL,
    price_jpy           INTEGER,                         -- NULL = データなし
    source_api          VARCHAR(20),
    cached_at           TIMESTAMPTZ    NOT NULL DEFAULT NOW(),
    expires_at          TIMESTAMPTZ    NOT NULL,          -- cached_at + interval '24 hours'

    UNIQUE (origin_iata, destination_iata, travel_date)
);

CREATE INDEX idx_pcc_route_month ON price_calendar_cache(origin_iata, destination_iata, travel_date);
CREATE INDEX idx_pcc_expires     ON price_calendar_cache(expires_at);
```

### 3.4 inspire_cache

「どこでも検索」（Amadeus Flight Inspirations）の結果キャッシュ。

```sql
CREATE TABLE inspire_cache (
    id                    UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    search_query_hash     VARCHAR(64)    NOT NULL UNIQUE,
    origin_iata           CHAR(3)        NOT NULL REFERENCES airports(iata_code),
    departure_date_from   DATE           NOT NULL,
    departure_date_to     DATE           NOT NULL,
    max_price_jpy         INTEGER,
    result_json           JSONB          NOT NULL,
    cached_at             TIMESTAMPTZ    NOT NULL DEFAULT NOW(),
    expires_at            TIMESTAMPTZ    NOT NULL           -- cached_at + interval '6 hours'
);

CREATE INDEX idx_ic_hash    ON inspire_cache(search_query_hash);
CREATE INDEX idx_ic_expires ON inspire_cache(expires_at);
```

### 3.5 sale_events

RSS・Trip.com Affiliate・Gemini グラウンディングから収集したセール情報。

```sql
CREATE TABLE sale_events (
    id                  UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    category            VARCHAR(20)    NOT NULL,  -- 'flight' | 'hotel' | 'package'
    title               VARCHAR(200)   NOT NULL,
    description         TEXT,
    sale_start          TIMESTAMPTZ,
    sale_end            TIMESTAMPTZ,
    travel_start        DATE,
    travel_end          DATE,
    discount_rate       DECIMAL(5, 2), -- 40.00 = 40% OFF
    min_price_jpy       INTEGER,       -- 目安最低価格（実際の予約価格ではない）
    target_routes       JSONB,         -- [{"origin":"HND","destination":"OKA"}, ...]
    target_destinations JSONB,         -- ["OKA", "CTS", ...]
    coupon_code         VARCHAR(50),
    booking_url         TEXT,
    source              VARCHAR(50)    NOT NULL,
    -- 'jal_rss' | 'ana_rss' | 'peach_rss' | 'jetstar_rss'
    -- | 'tripcom_affiliate' | 'gemini_grounding' | 'manual'
    source_url          TEXT,
    is_verified         BOOLEAN        NOT NULL DEFAULT FALSE,
    -- FALSE = Gemini取得（価格精度低）, TRUE = 公式RSS/API
    external_id         VARCHAR(200),  -- ソース側の ID（重複防止用）
    created_at          TIMESTAMPTZ    NOT NULL DEFAULT NOW(),
    updated_at          TIMESTAMPTZ    NOT NULL DEFAULT NOW(),

    UNIQUE (source, external_id)       -- 同一セールの重複登録防止
);

CREATE INDEX idx_se_sale_end   ON sale_events(sale_end);
CREATE INDEX idx_se_category   ON sale_events(category);
CREATE INDEX idx_se_source     ON sale_events(source);
CREATE INDEX idx_se_created    ON sale_events(created_at DESC);
```

### 3.6 favorites

お気に入り路線・ホテルの登録。個人利用のため user_id 不要。

```sql
CREATE TABLE favorites (
    id          UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    type        VARCHAR(20)    NOT NULL,    -- 'flight_route' | 'hotel'
    label       VARCHAR(100),              -- 表示名（例: "東京→バンコク"）
    params      JSONB          NOT NULL,
    -- flight_route: {"origin":"HND","destination":"BKK","cabin":"ECONOMY"}
    -- hotel:        {"city":"BKK","checkin":"2026-06-01","nights":3}
    created_at  TIMESTAMPTZ    NOT NULL DEFAULT NOW()
);
```

### 3.7 price_alerts

お気に入りに設定した価格アラート。

```sql
CREATE TABLE price_alerts (
    id                      UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    favorite_id             UUID           NOT NULL REFERENCES favorites(id) ON DELETE CASCADE,
    target_price_jpy        INTEGER        NOT NULL,
    last_checked_price_jpy  INTEGER,                   -- 直近チェック時の最安値
    last_notified_at        TIMESTAMPTZ,               -- 最終通知日時
    notify_method           VARCHAR(20)    NOT NULL DEFAULT 'push',
    is_active               BOOLEAN        NOT NULL DEFAULT TRUE,
    created_at              TIMESTAMPTZ    NOT NULL DEFAULT NOW(),
    updated_at              TIMESTAMPTZ    NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_pa_active ON price_alerts(is_active) WHERE is_active = TRUE;
```

### 3.8 push_subscriptions

Web Push 通知用のデバイス購読情報。

```sql
CREATE TABLE push_subscriptions (
    id          UUID           PRIMARY KEY DEFAULT gen_random_uuid(),
    endpoint    TEXT           NOT NULL UNIQUE,   -- ブラウザの Push サービス URL
    p256dh      TEXT           NOT NULL,          -- 暗号化キー
    auth        TEXT           NOT NULL,          -- 認証シークレット
    user_agent  VARCHAR(300),
    created_at  TIMESTAMPTZ    NOT NULL DEFAULT NOW(),
    last_used_at TIMESTAMPTZ
);
```

## 4. キャッシュ設計

| テーブル | TTL | 理由 |
|---------|-----|------|
| `flight_search_cache` | **30分** | 航空券価格は変動が激しい |
| `price_calendar_cache` | **24時間** | 傾向把握用途なので即時性不要 |
| `inspire_cache` | **6時間** | 目的地インスピレーションは安定 |

**クリーンアップ**: APScheduler（Cloud Scheduler から毎日 0時に起動）で `expires_at < NOW()` のレコードを削除する。

```python
# scheduler/jobs.py
async def cleanup_expired_cache():
    await db.execute(
        "DELETE FROM flight_search_cache WHERE expires_at < NOW()"
    )
    await db.execute(
        "DELETE FROM price_calendar_cache WHERE expires_at < NOW()"
    )
    await db.execute(
        "DELETE FROM inspire_cache WHERE expires_at < NOW()"
    )
    # 終了済みセールイベントは60日後に削除
    await db.execute(
        "DELETE FROM sale_events WHERE sale_end < NOW() - interval '60 days'"
    )
```

## 5. インデックス戦略

| 目的 | インデックス |
|------|------------|
| フライト検索キャッシュHIT | `flight_search_cache(search_query_hash)` |
| 有効期限チェック（クリーンアップ） | `*_cache(expires_at)` |
| カレンダー表示（路線×月） | `price_calendar_cache(origin, destination, travel_date)` |
| セールカレンダー表示 | `sale_events(sale_end)`, `sale_events(category)` |
| 価格アラート定期チェック | `price_alerts(is_active)` partial index |

## 8. SQLAlchemy モデル（概要）

```python
# models.py（抜粋）

class Airport(Base):
    __tablename__ = "airports"
    iata_code = Column(CHAR(3), primary_key=True)
    name_ja   = Column(String(100), nullable=False)
    name_en   = Column(String(100), nullable=False)
    city_ja   = Column(String(100))
    city_en   = Column(String(100))
    country_code = Column(CHAR(2), nullable=False)
    latitude  = Column(Numeric(9, 6))
    longitude = Column(Numeric(9, 6))

class FlightSearchCache(Base):
    __tablename__ = "flight_search_cache"
    id                = Column(UUID(as_uuid=True), primary_key=True, default=uuid4)
    search_query_hash = Column(String(64), unique=True, nullable=False)
    origin_iata       = Column(CHAR(3), ForeignKey("airports.iata_code"), nullable=False)
    destination_iata  = Column(CHAR(3), ForeignKey("airports.iata_code"), nullable=False)
    departure_date    = Column(Date, nullable=False)
    return_date       = Column(Date)
    result_json       = Column(JSONB, nullable=False)
    source_api        = Column(String(20), nullable=False)
    cached_at         = Column(TIMESTAMPTZ, default=func.now(), nullable=False)
    expires_at        = Column(TIMESTAMPTZ, nullable=False)

class SaleEvent(Base):
    __tablename__ = "sale_events"
    id                  = Column(UUID(as_uuid=True), primary_key=True, default=uuid4)
    category            = Column(String(20), nullable=False)
    title               = Column(String(200), nullable=False)
    description         = Column(Text)
    sale_start          = Column(TIMESTAMPTZ)
    sale_end            = Column(TIMESTAMPTZ)
    travel_start        = Column(Date)
    travel_end          = Column(Date)
    discount_rate       = Column(Numeric(5, 2))
    min_price_jpy       = Column(Integer)
    target_routes       = Column(JSONB)
    target_destinations = Column(JSONB)
    coupon_code         = Column(String(50))
    booking_url         = Column(Text)
    source              = Column(String(50), nullable=False)
    is_verified         = Column(Boolean, default=False, nullable=False)
    external_id         = Column(String(200))
    created_at          = Column(TIMESTAMPTZ, default=func.now(), nullable=False)
    updated_at          = Column(TIMESTAMPTZ, default=func.now(), onupdate=func.now(), nullable=False)

class Favorite(Base):
    __tablename__ = "favorites"
    id         = Column(UUID(as_uuid=True), primary_key=True, default=uuid4)
    type       = Column(String(20), nullable=False)
    label      = Column(String(100))
    params     = Column(JSONB, nullable=False)
    created_at = Column(TIMESTAMPTZ, default=func.now(), nullable=False)
    alerts     = relationship("PriceAlert", back_populates="favorite", cascade="all, delete-orphan")

class PriceAlert(Base):
    __tablename__ = "price_alerts"
    id                     = Column(UUID(as_uuid=True), primary_key=True, default=uuid4)
    favorite_id            = Column(UUID(as_uuid=True), ForeignKey("favorites.id", ondelete="CASCADE"), nullable=False)
    target_price_jpy       = Column(Integer, nullable=False)
    last_checked_price_jpy = Column(Integer)
    last_notified_at       = Column(TIMESTAMPTZ)
    is_active              = Column(Boolean, default=True, nullable=False)
    favorite               = relationship("Favorite", back_populates="alerts")

class PushSubscription(Base):
    __tablename__ = "push_subscriptions"
    id         = Column(UUID(as_uuid=True), primary_key=True, default=uuid4)
    endpoint   = Column(Text, unique=True, nullable=False)
    p256dh     = Column(Text, nullable=False)
    auth       = Column(Text, nullable=False)
    user_agent = Column(String(300))
    created_at = Column(TIMESTAMPTZ, default=func.now(), nullable=False)
```
